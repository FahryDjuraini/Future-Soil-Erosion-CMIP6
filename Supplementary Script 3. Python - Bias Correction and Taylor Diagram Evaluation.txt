def normalize_colname(c):
    c2 = re.sub(r'[^A-Za-z0-9]', '_', str(c).upper())
    c2 = re.sub(r'_+', '_', c2).strip('_')
    return c2

# METRIC FUNCTIONS
# ===============================
def rmse(o, s):
    o = np.array(o, dtype=float); s = np.array(s, dtype=float)
    return np.sqrt(np.nanmean((s - o)**2))

def pbias(o, s):
    o = np.array(o, dtype=float); s = np.array(s, dtype=float)
    return 100 * np.nansum(s - o) / (np.nansum(o) + 1e-12)

def kge(o, s):
    o = np.array(o, dtype=float); s = np.array(s, dtype=float)
    if np.nanstd(o) == 0 or np.nanstd(s) == 0:
        return np.nan
    cc = np.corrcoef(o, s)[0, 1]
    alpha = np.nanstd(s) / (np.nanstd(o) + 1e-12)
    beta  = np.nanmean(s) / (np.nanmean(o) + 1e-12)
    return 1 - np.sqrt((cc-1)**2 + (alpha-1)**2 + (beta-1)**2)

def to_num(s):
    return pd.to_numeric(s, errors="coerce")

def fit_qm_monthly_params(obs_series, mod_series, threshold=1.0):

    df = pd.DataFrame({"obs": obs_series.values,
                       "mod": mod_series.values},
                       index=obs_series.index)

    df["month"] = df.index.month
    month_models = {}

    for m in range(1,13):

        sub = df[df["month"]==m]

        if sub.empty:
            month_models[m] = {
                "method":"empirical",
                "params":{"p":np.array([0,1]),"q_obs":np.array([0,0])},
                "mean_obs":0.0,
                "std_obs":0.0
            }
            continue

        obs_m = to_num(sub["obs"]).copy()
        mod_m = to_num(sub["mod"]).copy()

        obs_m[obs_m<=0]=1e-4
        mod_m[mod_m<=0]=1e-4

        obs_pos = obs_m[obs_m>=threshold]
        mod_pos = mod_m[mod_m>=threshold]

        mean_obs=float(np.nanmean(obs_m))
        std_obs=float(np.nanstd(obs_m))

        if len(obs_pos)>=min_pos_len and len(mod_pos)>=min_pos_len:

            mean_o,var_o=np.mean(obs_pos),np.var(obs_pos)
            mean_m,var_m=np.mean(mod_pos),np.var(mod_pos)

            if mean_o>0 and var_o>0 and mean_m>0 and var_m>0:

                O_alpha=(mean_o**2)/var_o
                O_beta=var_o/mean_o

                M_alpha=(mean_m**2)/var_m
                M_beta=var_m/mean_m

                month_models[m]={
                    "method":"gamma",
                    "params":{
                        "O_alpha":float(O_alpha),
                        "O_beta":float(O_beta),
                        "M_alpha":float(M_alpha),
                        "M_beta":float(M_beta)
                    },
                    "mean_obs":mean_obs,
                    "std_obs":std_obs
                }

                continue

        q_obs=np.quantile(obs_m,quant_grid)

        month_models[m]={
            "method":"empirical",
            "params":{
                "p":quant_grid,
                "q_obs":q_obs.astype(float)
            },
            "mean_obs":mean_obs,
            "std_obs":std_obs
        }

    return month_models


def apply_qm_to_series(series,month_models,threshold=1.0):

    s=pd.Series(to_num(series).values,index=series.index)
    s_fill=s.copy()

    s_fill[s_fill<=0]=1e-4

    months=s_fill.index.month
    corrected=np.zeros_like(s_fill.values,float)

    for m in range(1,13):

        mask=(months==m)
        if not mask.any():continue

        vals=s_fill[mask].values
        raw=s[mask].values
        model=month_models[m]

        if model["method"]=="gamma":

            par=model["params"]

            F=gamma.cdf(vals,
                        par["M_alpha"],
                        scale=par["M_beta"])

            x_corr=gamma.ppf(
                np.clip(F,1e-6,1-1e-6),
                par["O_alpha"],
                scale=par["O_beta"]
            )

        else:

            p=model["params"]["p"]
            q_obs=model["params"]["q_obs"]

            F=rankdata(vals,method="average")/(len(vals)+1.0)

            x_corr=np.interp(F,p,q_obs)

        x_corr=np.where(np.isnan(x_corr),vals,x_corr)
        x_corr[x_corr<0]=0.0

        mean_obs=model["mean_obs"]
        std_obs=model["std_obs"]

        if std_obs>0:

            mc,sc=np.nanmean(x_corr),np.nanstd(x_corr)

            if sc>0:
                x_corr=(x_corr-mc)*(std_obs/sc)+mean_obs

            mc2=np.nanmean(x_corr)

            if mc2!=0:
                x_corr*=mean_obs/(mc2+1e-12)

        x_corr[raw<threshold]=0.0

        corrected[mask]=x_corr

    return pd.Series(corrected,index=series.index,name="QM")

def mme_statistics(df,models):

    arr=np.array(df[models].values,float)

    mme_mean=np.nanmean(arr,axis=1)
    mme_median=np.nanmedian(arr,axis=1)

    return {
        "MME_MEAN":pd.Series(mme_mean,index=df.index),
        "MME_MEDIAN":pd.Series(mme_median,index=df.index)
    }
